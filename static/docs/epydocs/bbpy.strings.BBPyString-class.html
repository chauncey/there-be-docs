A string with template applied

>>> env = BBPyString( "6559", "remote_user:C(4)" )
>>> env.remote_user

>>> env['remote_user']
u'6559'
>>> env['REMOTE_USER']
u'6559'
>>> env.get('remote_user','abc')
u'6559'
>>> env.get('badfield','abc')
